1. Program -> Declaration-list
2. Declaration-list -> Declaration Declaration-list | EPSILON
3. Declaration -> Declaration-initial Declaration-prime
done 4. Declaration-initial -> #ptype Type-specifier #pid_declare ID
5. Declaration-prime -> Fun-declaration-prime | Var-declaration-prime
done 6. Var-declaration-prime -> #declare_var; | [ #pnum NUM ] #declare_array;
done 7. Fun-declaration-prime -> #declare_func ( Params ) Compound-stmt
done 8. Type-specifier -> int | void
done 9. Params -> #ptype int #pid_declare ID Param-prime Param-list | void
done 10. Param-list -> , Param Param-list | EPSILON
done 11. Param -> Declaration-initial Param-prime
done 12. Param-prime -> [ ] #param_arr | EPSILON #param_var
done 13. Compound-stmt -> { Declaration-list Statement-list }
done 14. Statement-list -> Statement Statement-list | EPSILON
done 15. Statement -> Expression-stmt | Compound-stmt | Selection-stmt | Iteration-stmt | Return-stmt
done 16. Expression-stmt -> Expression ; | break #brek; | ;
done 17. Selection-stmt -> if ( Expression ) #save Statement Else-stmt
done 18. Else-stmt -> endif #jpf | #jpf_save else Statement endif #jp
done 19. Iteration-stmt -> for (Expression; #pb_save Expression #save #save; #pb_save Expression #save) #for_start #pb_save Statement #fora
done 20. Return-stmt -> return Return-stmt-prime #set_ra
done 21. Return-stmt-prime -> ; | Expression #set_rv ;
done 22. Expression -> Simple-expression-zegond | #pid ID B
done 23. B -> = Expression #assign | [ Expression ] #parr_idx H | Simple-expression-prime
done 24. H -> = Expression #assign | G D C
done 25. Simple-expression-zegond -> Additive-expression-zegond C
done 26. Simple-expression-prime -> Additive-expression-prime C
done 27. C -> Relop Additive-expression #relop | EPSILON
done 28. Relop -> #p_op < | #p_op ==
done 29. Additive-expression -> Term D
done 30. Additive-expression-prime -> Term-prime D
done 31. Additive-expression-zegond -> Term-zegond D
done 32. D -> Addop Term #add D | EPSILON
done 33. Addop -> #p_op + | #p_op -
done 34. Term -> Signed-factor G
done 35. Term-prime -> Signed-factor-prime G
done 36. Term-zegond -> Signed-factor-zegond G
done 37. G -> #p_op * Signed-factor #mult G | EPSILON
38. Signed-factor -> #p_op + Factor #add | #p_op - Factor #add | Factor
39. Signed-factor-prime -> Factor-prime
40. Signed-factor-zegond -> #p_op + Factor #add | #p_op - Factor #add | Factor-zegond
done 41. Factor -> ( Expression ) | #pid ID Var-call-prime | #pnum NUM
done 42. Var-call-prime -> ( Args ) #call_func | Var-prime
done 43. Var-prime -> [ Expression ] #parr_idx | EPSILON
done 44. Factor-prime -> ( Args ) #call_func | EPSILON
done 45. Factor-zegond -> ( Expression ) | #pnum NUM
done 46. Args -> Arg-list | EPSILON
done 47. Arg-list -> Expression Arg-list-prime
done 48. Arg-list-prime -> , Expression Arg-list-prime | EPSILON
